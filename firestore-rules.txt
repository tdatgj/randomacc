// Firestore Security Rules cho project randomacc-96218
// Copy rules này vào Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection - users chỉ có thể đọc/ghi dữ liệu của chính họ
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Transactions collection - users chỉ có thể đọc/ghi giao dịch của chính họ
    match /transactions/{transactionId} {
      allow read, write: if request.auth != null && 
        request.auth.uid == resource.data.userId;
      allow create: if request.auth != null && 
        request.auth.uid == request.resource.data.userId;
    }
    
    // Cho phép đọc tất cả users để tìm kiếm theo username (cho WebHook)
    match /users/{userId} {
      allow read: if true; // WebHook cần đọc để tìm user theo username
      allow write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
